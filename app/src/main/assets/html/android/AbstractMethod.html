📌 Java 示例代码


// BaseApp.java
abstract class BaseApp {
    BaseApp() {
        System.out.println("父类构造开始");
        System.out.println("抽象方法返回: " + getAppName());
        System.out.println("父类构造结束");
    }
    abstract String getAppName();
}

// MyApp.java
class MyApp extends BaseApp {
    @Override
    String getAppName() {
        return "My Application";
    }
}
