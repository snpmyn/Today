Activity 生命周期

onCreate()   # Activity 创建并初始化 UI
onStart()    # Activity 可见
onResume()   # Activity 前台运行，可交互

onPause()    # Activity 失去焦点
onStop()     # Activity 完全不可见
onDestroy()  # Activity 销毁

onRestart()  # 从 Stop 状态重新启动
