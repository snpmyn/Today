🌸 Git 指令速查表 🌸
配置 常用操作 分支 远程 暂存 标签 查看 撤销

配置

git config --global user.name "你的名字"
git config --global user.email "你的邮箱"
git config --list        # 查看所有配置
git help 指令            # 查看帮助

常用操作

git init                 # 初始化仓库
git clone URL            # 克隆远程仓库
git status               # 查看当前状态
git add 文件/目录         # 添加到暂存区
git add .                # 添加所有更改
git commit -m "说明"     # 提交更改
git commit --amend       # 修改最近一次提交
git rm 文件名            # 删除文件并提交
git mv 旧名 新名         # 文件改名

分支操作

git branch               # 查看分支
git branch 分支名         # 创建分支
git checkout 分支名       # 切换分支
git checkout -b 分支名    # 创建并切换分支
git merge 分支名          # 合并分支
git rebase 分支名         # 变基合并
git branch -d 分支名      # 删除分支
git branch -D 分支名      # 强制删除分支

远程仓库

git remote -v                   # 查看远程仓库
git remote add origin URL        # 添加远程仓库
git remote remove origin         # 删除远程
git push -u origin 分支名        # 推送到远程
git push origin --delete 分支名  # 删除远程分支
git pull origin 分支名           # 拉取并合并远程更新
git fetch origin                 # 获取远程更新但不合并

暂存与恢复

git stash save "说明"     # 保存当前修改
git stash list            # 查看所有 stash
git stash apply           # 应用最近一次 stash
git stash pop             # 应用并删除最近一次 stash
git stash drop            # 删除某个 stash
git stash clear           # 清空所有 stash

标签操作

git tag                  # 查看标签
git tag 标签名            # 新建标签
git tag -a 标签名 -m "说明" # 新建带注释标签
git push origin 标签名     # 推送标签
git push origin --tags    # 推送所有标签
git tag -d 标签名          # 删除本地标签
git push origin :refs/tags/标签名 # 删除远程标签

查看与搜索

git log                  # 查看提交历史
git log --oneline        # 简洁日志
git log --graph --oneline --decorate --all  # 图形化查看
git show 提交ID           # 查看某次提交详情
git diff                 # 查看工作区修改
git diff --cached        # 查看暂存区修改
git grep 关键词           # 搜索仓库内内容

撤销与回退

git checkout -- 文件名   # 撤销未暂存修改
git reset 文件名          # 从暂存区撤回文件
git reset --soft 提交ID   # 回退到指定提交，保留修改
git reset --hard 提交ID   # 回退到指定提交，丢弃修改
git revert 提交ID         # 撤销指定提交（生成新提交）
git clean -fd             # 删除未跟踪文件/目录
